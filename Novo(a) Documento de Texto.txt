#include <stdio.h>
#include <math.h>

int main()
{
    float num,num2,num3;
    printf("Digite tres numeros aleatorios, podendo ser eles negativo e positivo\n");
    scanf("%f %f %f", &num,&num2,&num3);
    if (num > 0){
     printf("\nA raiz do seu numero é: %.2f", sqrt(num));
    }
    else if (num < 0){
     printf("\nO quadrado do seu numero é: %.2f", num*num);
    }
    if (num2 >= 10 && num2 <= 100){ 
        printf("\nO seu numero esta entre 10 e 100, intervalo permitido");
    }
    else if(num2 < 10 && num2 > 100){
        printf("\nO seu numero é menor que 10 ou maior que 100, intervalo não permitido: %.2f", num2);
    }
    if (num3 < num2){
        printf("\nA diferença entre os numeros 1 e 2 é de: %.2f", num3-num2);
    }
    else if(num3 > num2){
        printf("\nO resultado da soma do numero 3 com + 1 é: %.2f", num3+1);
    }
    

    return 0;
}
